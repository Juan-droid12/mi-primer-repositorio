2. definir valor de pi
3. otener valores de altura y radio
4. calcular valor del volumen utilizando la ecuaci√≥n

pseudocodigo
program calcular_vol_cono
Start
//programa para calcular volumen de un cubo 
// declaracion de variables
constant float PI = 3.1416
declare float radius
declare float altura
declare float volumen 

display "ingrese valor de radio"
input radius 
display "ingrese la altura del cono"
input altura
set volumen = (PI * radius ^ 2 * altura ) / 3
display "el volumen del cono es ", volumen
End
